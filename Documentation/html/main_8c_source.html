205void init_digital_pins(void) {
206 PCC->PCCn[PCC_PORTC_INDEX] |= 0x40000000; // Write Clock Gate Control high to allow updating registers
207 PCC->PCCn[PCC_PORTB_INDEX] |= 0x40000000;
208
209 // Set Pin Control Registers
210 PORTB->PCR[0] = 0x00000103; // Puts pin in alternate mode 1 (GPIO use), enables and sets internal pull up resistor
211 PORTC->PCR[0] = 0x00000103;
212 PORTC->PCR[14] = 0x00000103;
213 PORTC->PCR[16] = 0x00000103;
214
215 // Set Port Data Direction Register for inputs
216 PTB->PDDR &=~ (0x1);
217 PTC->PDDR &=~ ((0x1 << 16) | (0x1 << 14) | 0x1);
218}

257void debounce_analog_contact(struct analog_contact *a) {
258 uint16_t v = 0; // Stores the return value from the read_adc function
259
260 if (a->adc == 0) {
261 v = read_adc0(a->ch);
262 }
263 else {
264 v = read_adc1(a->ch);
265 }
266
267 if (v < FH && v > FL && a->state != press) { // If state is pressed
268 a->press += 1;
269 a->unpress = 0;
270
271 if (a->press >= DL) { // Update state when debounce limit is reached
272 a->state = press;
273 a->time = count; // Set time equal to current time since startup whenever state changes
274 }
275 }
276 else if (v < IH && v > IL && a->state != open) { // If state is unpressed
277 a->press = 0;
278 a->unpress += 1;
279
280 if (a->unpress >= DL) {
281 a->state = open;
282 a->time = count;
283 }
284 }
285 else if (v < IL && v > FH && a->state != transition) { // If state is transition
286 if (a->transition_count <= 0) // If the transition count is 0 start counting to 2500 samples
287 a->transition_count = 1;
288
289 a->press = 0;
290 a->unpress = 0;
291 a->transition += 1;
292 }
293 else if (v > IH && a->state != high) { // If state is invalid high
294 if (a->high_count <= 0)
295 a->high_count = 1;
296
297 a->invalid_high += 1;
298 a->press = 0;
299 a->unpress = 0;
300 }
301 else if (v < FL && a->state != low) { // If state is invalid low
302 if (a->low_count <= 0)
303 a->low_count = 1;
304
305 a->invalid_low += 1;
306 a->press = 0;
307 a->unpress = 0;
308 }
309
310 // Handles the 2000/2500 samples requirement for the transition region
311 if (a->transition_count > 0 && a->state != transition) { // If transition count has been started, increment every call to func
312 a->transition_count += 1;
313
314 if (a->transition_count >= 2500) {
315 if (a->transition >= 2000) {
316 a->state = transition; // If 2000 of last 2500 samples were transition update state
317 a->time = count;
318 }
319 // After 2500 samples, reset the transition variables
320 a->transition = 0;
321 a->transition_count = 0;
322 }
323 }
324
325 // Handles 16/20 sample requirement for invalid low state
326 if (a->low_count > 0 && a->state != low) {
327 a->low_count += 1;
328
329 if (a->low_count >= 20) {
330 if (a->invalid_low >= 16) {
331 a->state = low; // If 16 of last 20 samples were invalid low update state
332 a->time = count;
333 }
334 // After 20 samples, reset the variables
335 a->invalid_low = 0;
336 a->low_count = 0;
337 }
338 }
339
340 // Handles 16/20 sample requirement for invalid high state
341 if (a->high_count > 0 && a->state != high) {
342 a->high_count += 1;
343
344 if (a->high_count >= 20) {
345 if (a->invalid_high >= 16) {
346 a->state = high; // If 16 of last 20 samples were invalid high update state
347 a->time = count;
348 }
349 // After 20 samples, reset the variables
350 a->invalid_high = 0;
351 a->high_count = 0;
352 }
353 }
354}

367void debounce_btn(struct btn *b, uint32_t digital_condition) {
368 // Check analog switches
369 debounce_analog_contact(&(b->A1));
370 debounce_analog_contact(&(b->A2));
371
372 // Check digital switch
373 if (digital_condition == 0 && b->D1.state != press) { // If state is pressed
374 b->D1.press += 1;
375 b->D1.unpress = 0;
376 if (b->D1.press >= DL) {
377 b->D1.state = press;
378 b->D1.time = count;
379 }
380 }
381 else if (digital_condition != 0 && b->D1.state != open) { // If state is unpressed
382 b->D1.press = 0;
383 b->D1.unpress += 1;
384 if (b->D1.unpress >= DL) {
385 b->D1.state = open;
386 b->D1.time = count;
387 }
388 }
389
390 // Determine final debounced state based on the states of the three contacts
391 if (b->A1.state == b->A2.state && b->A1.state == b->D1.state) {
392 b->state = b->A1.state; // If all three states are the same that as the overall button state
393 }
394
395// printf("Analog 1: %d,\tAnalog 2: %d,\tDigital 1: %d\n", b->A1.state, b->A2.state, b->D1.state); // DEBUG USE ONLY
396
397}

141uint16_t read_adc0(uint8_t ch) {
142 uint32_t ADC_RawResult;
143
144 ADC0->SC1[0] = ADC_SC1_ADCH(ch); // Initiate new conversion by writing to ADC0_SC1A(ADCH)
145
146 while(((ADC0->SC1[0] & ADC_SC1_COCO_MASK) >> ADC_SC1_COCO_SHIFT) == 0); // Wait for latest conversion to complete
147 ADC_RawResult = ADC0->R[0]; // Read ADC Data Result A (ADC0_RA)
148 return (ADC_RawResult * 5000) / ((1 << 10) - 1); // Convert to mV (@VREFH = 5V)

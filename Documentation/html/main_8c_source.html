248void init_digital_pins(void) {
249 PCC->PCCn[PCC_PORTC_INDEX] |= 0x40000000; // Write Clock Gate Control high to allow updating registers
250 PCC->PCCn[PCC_PORTB_INDEX] |= 0x40000000;
251
252 // Set Pin Control Registers
253 PORTB->PCR[0] = 0x00000103; // Puts pin in alternate mode 1 (GPIO use), enables and sets internal pull up resistor
254 PORTC->PCR[0] = 0x00000103;
255 PORTC->PCR[14] = 0x00000103;
256 PORTC->PCR[16] = 0x00000103;
257
258 // Set Port Data Direction Register for inputs
259 PTB->PDDR &=~ (0x1);
260 PTC->PDDR &=~ ((0x1 << 16) | (0x1 << 14) | 0x1);
261}

300void debounce_analog_contact(struct analog_contact *a) {
301 uint16_t v = 0; // Stores the return value from the read_adc function
302
303 if (a->adc == 0) {
304 v = read_adc0(a->ch);
305 }
306 else {
307 v = read_adc1(a->ch);
308 }
309
310 if (v < FH && v > FL && a->state != press) { // If state is pressed
311 a->press += 1;
312 a->unpress = 0;
313
314 if (a->press >= DL) { // Update state when debounce limit is reached
315 a->state = press;
316 a->time = count; // Set time equal to current time since startup whenever state changes
317 }
318 }
319 else if (v < IH && v > IL && a->state != open) { // If state is unpressed
320 a->press = 0;
321 a->unpress += 1;
322
323 if (a->unpress >= DL) {
324 a->state = open;
325 a->time = count;
326 }
327 }
328 else if (v < IL && v > FH && a->state != transition) { // If state is transition
329 if (a->transition_count <= 0) // If the transition count is 0 start counting to 2500 samples
330 a->transition_count = 1;
331
332 a->press = 0;
333 a->unpress = 0;
334 a->transition += 1;
335 }
336 else if (v > IH && a->state != high) { // If state is invalid high
337 if (a->high_count <= 0)
338 a->high_count = 1;
339
340 a->invalid_high += 1;
341 a->press = 0;
342 a->unpress = 0;
343 }
344 else if (v < FL && a->state != low) { // If state is invalid low
345 if (a->low_count <= 0)
346 a->low_count = 1;
347
348 a->invalid_low += 1;
349 a->press = 0;
350 a->unpress = 0;
351 }
352
353 // Handles the 2000/2500 samples requirement for the transition region
354 if (a->transition_count > 0 && a->state != transition) { // If transition count has been started, increment every call to func
355 a->transition_count += 1;
356
357 if (a->transition_count >= 2500) {
358 if (a->transition >= 2000) {
359 a->state = transition; // If 2000 of last 2500 samples were transition update state
360 a->time = count;
361 }
362 // After 2500 samples, reset the transition variables
363 a->transition = 0;
364 a->transition_count = 0;
365 }
366 }
367
368 // Handles 16/20 sample requirement for invalid low state
369 if (a->low_count > 0 && a->state != low) {
370 a->low_count += 1;
371
372 if (a->low_count >= 20) {
373 if (a->invalid_low >= 16) {
374 a->state = low; // If 16 of last 20 samples were invalid low update state
375 a->time = count;
376 }
377 // After 20 samples, reset the variables
378 a->invalid_low = 0;
379 a->low_count = 0;
380 }
381 }
382
383 // Handles 16/20 sample requirement for invalid high state
384 if (a->high_count > 0 && a->state != high) {
385 a->high_count += 1;
386
387 if (a->high_count >= 20) {
388 if (a->invalid_high >= 16) {
389 a->state = high; // If 16 of last 20 samples were invalid high update state
390 a->time = count;
391 }
392 // After 20 samples, reset the variables
393 a->invalid_high = 0;
394 a->high_count = 0;
395 }
396 }
397}

410void debounce_btn(struct btn *b, uint32_t digital_condition) {
411 // Check analog switches
412 debounce_analog_contact(&(b->A1));
413 debounce_analog_contact(&(b->A2));
414
415 // Check digital switch
416 if (digital_condition == 0 && b->D1.state != press) { // If state is pressed
417 b->D1.press += 1;
418 b->D1.unpress = 0;
419 if (b->D1.press >= DL) {
420 b->D1.state = press;
421 b->D1.time = count;
422 }
423 }
424 else if (digital_condition != 0 && b->D1.state != open) { // If state is unpressed
425 b->D1.press = 0;
426 b->D1.unpress += 1;
427 if (b->D1.unpress >= DL) {
428 b->D1.state = open;
429 b->D1.time = count;
430 }
431 }
432
433 // Determine final debounced state based on the states of the three contacts
434 if (b->A1.state == b->A2.state && b->A1.state == b->D1.state) {
435 b->last = b->state;
436 b->state = b->A1.state; // If all three states are the same that as the overall button state
437 }
438
439// printf("Analog 1: %d,\tAnalog 2: %d,\tDigital 1: %d\n", b->A1.state, b->A2.state, b->D1.state); // DEBUG USE ONLY
440
441}

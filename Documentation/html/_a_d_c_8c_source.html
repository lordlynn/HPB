111uint16_t read_adc0(uint8_t ch) {
112 uint32_t ADC_RawResult;
113 // Initiate new conversion by writing to ADC0_SC1A(ADCH)
114 ADC0->SC1[0] = ADC_SC1_ADCH(ch);
115
116 // Wait for latest conversion to complete

118 ADC_RawResult = ADC0->R[0]; // Read ADC Data Result A (ADC0_RA)

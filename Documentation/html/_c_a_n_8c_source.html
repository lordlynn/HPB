11void CAN0_init(void) {
12 int i = 0;
13 PCC->PCCn[PCC_FlexCAN0_INDEX] |= PCC_PCCn_CGC_MASK; // CGC=1: enable clock to FlexCAN0
14 CAN0->MCR |= CAN_MCR_MDIS_MASK; // MDIS=1: Disable module before selecting clock
15 CAN0->CTRL1 |= CAN_CTRL1_CLKSRC_MASK; // CLKSRC=1: Clock Source = Peripheral clock (80MHz)
16 CAN0->MCR &= ~CAN_MCR_MDIS_MASK; // MDIS=0; Enable module config. (Sets FRZ, HALT)
17
18 while (!((CAN0->MCR & CAN_MCR_FRZACK_MASK) >> CAN_MCR_FRZACK_SHIFT)); // Good practice: wait for FRZACK=1 on freeze mode entry/exit
19
20 CAN0->CTRL1 = 0; // Timing variables set in CBT so nothing from example to set here
21 CAN0->CTRL2 = (1 << 12); // Use CANFD standard ISO 11898-1
22 // CLK is 16MHz -> TQ = 62.5ns -> 1 / (500Kbits/s * 62.5ns) = 32
23 // So to set bit rate at 500k use 32 tq per bit.
24 // synch = 1, prop = 12, seg1 = 12, seg2 = 7
25 CAN0->CBT = 0x806139C7;
26
27 CAN0->FDCTRL = 0x000F0000; // Bit rate switching initially off, 64 data bytes,TDC disabled
28 BRS = 0; // BRS flag is true when BRS is enabled
29
30 // CLK is 8MHz -> TQ = 62.5ns -> 1 / (2Mbits/s * 62.5ns) = 8
31 // So to set bit rate at 2Mbits/s use 8 tq per bit.
32 // synch = 1, prop = 3, seg1 = seg2 = 2
33 CAN0->FDCBT = 0x00310C41;
34
35 for(i=0; i < 128; i++ ) { // CAN0: clear all 128 RAM words
36 CAN0->RAMn[i] = 0;
37 }
38
39 for(i=0; i < 16; i++ ) { // In FRZ mode, init CAN0 16 msg buf filters
40 CAN0->RXIMR[i] = 0xFFFFFFFF; // Check all ID bits for incoming messages
41 }
42
43 CAN0->RXMGMASK = 0x1FFFFFFF; // Global acceptance mask: check all ID bits
44
45 // Setup message buffer for to receive messages
46 CAN0->RAMn[4*MSG_BUF_SIZE + 0] = 0x04000000; // Msg Buf 4, word 0: Enable for reception
47 CAN0->RAMn[5*MSG_BUF_SIZE + 0] = 0x04000000;
48 /* EDL,BRS,ESI=0: CANFD not used */
49 /* CODE=4: MB set to RX inactive */
50 /* IDE=0: Standard ID */
51 /* SRR, RTR, TIME STAMP = 0: not applicable */
52
53 CAN0->RAMn[4*MSG_BUF_SIZE + 1] = (0x100 | DEVICE_ID) << 18; // Msg Buf 4, word 1: Standard ID = 0x01xx
54 CAN0->RAMn[5*MSG_BUF_SIZE + 1] = (0x200 | DEVICE_ID) << 18;
55
56
57 CAN0->MCR = 0x0000081F; // Set Last Message Buffer limit and enable CAN FD, exits freeze mode
58 while ((CAN0->MCR && CAN_MCR_FRZACK_MASK) >> CAN_MCR_FRZACK_SHIFT); // Good practice: wait for FRZACK to clear (not in freeze mode)
59 while ((CAN0->MCR && CAN_MCR_NOTRDY_MASK) >> CAN_MCR_NOTRDY_SHIFT); // Good practice: wait for NOTRDY to clear (module ready)
60
61 port_init();
62}

92void CAN0_transmit_msg(const unsigned char data[], int len, uint8_t msg_type) { // Assumption: Message buffer CODE is INACTIVE
93 CAN0->IFLAG1 |= 0x00000001; // Clear CAN 0 MB 0 flag without clearing others
94 uint8_t i, j, k;
95 uint8_t DLC_lengths[] = {0, 1, 2, 3, 4, 5, 6, 7, 8, 12, 16, 20, 24, 32, 48, 64};
96 uint8_t DLC;
97 uint8_t len_diff;
98
99 // Based on length of data array, determines which DLC setting to use
100 for (i = 0; i < 16; i++) {
101 CAN0->RAMn[0*MSG_BUF_SIZE + 2 + i] = 0x00000000; // Reset every data word in the message buffer
102 if (len <= DLC_lengths[i]) {
103 DLC = i;
104 len_diff = DLC_lengths[DLC] - len;
105 break;
106 }
107 }
108
109 for (i = 0; i < (len / 4); i++) { // Each word is 4 bytes. Write each word except the last into the message buffer
110 CAN0->RAMn[0*MSG_BUF_SIZE + 2 + i] = (data[i * 4 + 0] << 24) | (data[i * 4 + 1] << 16) | (data[i * 4 + 2] << 8) | (data[i * 4 + 3] << 0);
111 }
112
113 // Write the last bytes into the final message buffer
114 for (j = 0; j < len % 4; j++) {
115 CAN0->RAMn[0*MSG_BUF_SIZE + 2 + i] |= (data[i * 4 + j] << (24-j*8));
116 }
117
118 // fill empty bytes with '0' to matc DLC size
119 // j is the ending index of the data, so when j + k reaches 4 need to move to next word in message buffer
120 for (k = 0; k < len_diff; k++) {
121 CAN0->RAMn[0*MSG_BUF_SIZE + 2 + i + ((j + k) / 4)] |= '0' << (24 - ((j + k) % 4) * 8);
122 }
123
124 CAN0->RAMn[0*MSG_BUF_SIZE + 1] = ((msg_type << 8) | DEVICE_ID) << 18; // MB0 word 1: Tx msg with ID = msg_type << 8 | DEVICE ID
125
126 if (FD) {
127 CAN0->RAMn[0*MSG_BUF_SIZE + 0] = 0x8C400000 | (DLC << CAN_WMBn_CS_DLC_SHIFT) | (BRS << 30);
128 // EDL = 1 - Distinguishes between CAN and CAN FD
129 // BRS = 1 or 0 - Enable BRS
130 // ESI, RES = 0
131 // CODE = 0xC
132 // RES = 0
133 // SRR = 1 - substitute remote request
134 // IDE = 0 - No ID extension
135 // RTR = 0
136 // DLC = variable, Max of 0xF = 64 data bytes
137 }
138 else {
139 CAN0->RAMn[0*MSG_BUF_SIZE + 0] = 0x0C400000 | (DLC << CAN_WMBn_CS_DLC_SHIFT);
140 // EDL, BRS, ESI, RES = 0: CANFD not used
141 // CODE = 0xC: Activate msg buf to transmit
142 // SRR = 1 Tx frame (not req'd for std ID)
143 // IDE = 0: Standard ID
144 // RTR = 0: data, not remote tx request frame
145 // DLC = 8: 8-bytes
146 }
147
148 /* RAM0 of a message buffer used for Tx is structured as follows:
149 * EDL - 31
150 * BRS - 30
151 * ESI - 29
152 * RES - 28
153 * CODE - 24-27
154 * RES - 23
155 * SRR - 22
156 * IDE - 21
157 * RTR - 20
158 * DLC - 16-19
159 * TIME - 0-15
160 */
161
162}

165void CAN0_receive_msg(int buff) { // Receive msg from ID 0x556 using msg buffer 4
166 uint8_t j;
167 uint32_t dummy;
168 uint8_t DLC_lengths[] = {0, 1, 2, 3, 4, 5, 6, 7, 8, 12, 16, 20, 24, 32, 48, 64};
169
170 RxCODE = (CAN0->RAMn[ buff*MSG_BUF_SIZE + 0] & 0x07000000) >> 24; // Read CODE field
171 RxID = (CAN0->RAMn[ buff*MSG_BUF_SIZE + 1] & CAN_WMBn_ID_ID_MASK) >> CAN_WMBn_ID_ID_SHIFT ;
172 RxLENGTH = (CAN0->RAMn[ buff*MSG_BUF_SIZE + 0] & CAN_WMBn_CS_DLC_MASK) >> CAN_WMBn_CS_DLC_SHIFT;
173
174 RxLENGTH = DLC_lengths[RxLENGTH];
175
176 for (j = 0; j < RxLENGTH; j++) { // Read two words of data (8 bytes)
177 RxDATA[j] = CAN0->RAMn[ buff*MSG_BUF_SIZE + 2 + (j / 4)] >> (24 - j * 8);
178 }
179
180 RxTIMESTAMP = (CAN0->RAMn[ 0*MSG_BUF_SIZE + 0] & 0x000FFFF);
181 dummy = CAN0->TIMER; // Read TIMER to unlock message buffers
182 CAN0->IFLAG1 |= 0x1 << buff; // Clear CAN 0 message buffer flag without clearing others
183}

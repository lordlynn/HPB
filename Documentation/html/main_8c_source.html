438int is_valid_press(struct btn *b, uint32_t *time) {
439 uint32_t times[2];
440
441 /* Check that the digital contact and the analog contacts have the
442 * same state before calculating the delta time between their state
443 * changes. The delta time makes no sense unless at least two contacts
444 * have closed, and if that is the case only the shortest matters
445 * since it only takes two contacts to make a good press/open */
446 if (b->D1.state == b->A1.state)
447 times[0] = abs(b->A1.time - b->D1.time);
448 else
449 times[0] = 0xFFFFFFFF; // If no time is used for these contacts set the variable to 32bit int max
450 if (b->D1.state == b->A2.state)
451 times[1] = abs(b->A2.time - b->D1.time);
452 else
453 times[1] = 0xFFFFFFFF;
454
455 /* Digital contact + 1 analog contact is required for debounce,
456 * thus timeout failure occurs when neither analog contact
457 * debounces within the timeout limit to the digitial contact.
458 *
459 * If the delta time is reported over CAN as all Fs then ignore it */
460 if (times[0] < times[1])
461 *time = times[0];
462 else
463 *time = times[1];
464
465 /* Check for invalid press state. i.e. digital contact required and
466 * at least 1 analog contact. Check for invalid open states */
467 if ((b->D1.state != press || (b->A1.state != press && b->A2.state != press)) &&
468 (b->D1.state != open || (b->A1.state != open && b->A2.state != open)))
469 return 0;
470
471
472 if (times[0] < BTN_TIMEOUT || times[1] < BTN_TIMEOUT) {
473 return 1;
474 }
475 return 0;
476}

487void debounce_analog_contact(struct analog_contact *a) {
488 uint16_t v = 0; // Stores the return value from the read_adc function
489
490 if (a->adc == 0) {
491 v = read_adc0(a->ch);
492 }
493 else {
494 v = read_adc1(a->ch);
495 }
496 a->voltage = v;
497
498 if (v < FH && v > FL && a->state != press) { // If state is pressed
499 a->press += 1;
500 a->unpress = 0;
501
502 if (a->press >= DL) { // Update state when debounce limit is reached
503 a->state = press;
504 a->time = millis; // Set time equal to current time since startup whenever state changes
505 }
506 }
507 else if (v < IH && v > IL && a->state != open) { // If state is unpressed
508 a->press = 0;
509 a->unpress += 1;
510
511 if (a->unpress >= DL) {
512 a->state = open;
513 a->time = millis;
514 }
515 }
516 else if (v < IL && v > FH && a->state != transition) { // If state is transition
517 if (a->transition_count <= 0) // If the transition count is 0 start counting to 2500 samples
518 a->transition_count = 1;
519
520 a->press = 0;
521 a->unpress = 0;
522 a->transition += 1;
523 }
524 else if (v > IH && a->state != high) { // If state is invalid high
525 if (a->high_count <= 0)
526 a->high_count = 1;
527
528 a->invalid_high += 1;
529 a->press = 0;
530 a->unpress = 0;
531 }
532 else if (v < FL && a->state != low) { // If state is invalid low
533 if (a->low_count <= 0)
534 a->low_count = 1;
535
536 a->invalid_low += 1;
537 a->press = 0;
538 a->unpress = 0;
539 }
540
541 // Handles the 2000/2500 samples requirement for the transition region
542 if (a->transition_count > 0 && a->state != transition) { // If transition count has been started, increment every call to func
543 a->transition_count += 1;
544
545 if (a->transition_count >= 2500) {
546 if (a->transition >= 2000) {
547 a->state = transition; // If 2000 of last 2500 samples were transition update state
548 a->time = millis;
549 }
550 // After 2500 samples, reset the transition variables
551 a->transition = 0;
552 a->transition_count = 0;
553 }
554 }
555
556 // Handles 16/20 sample requirement for invalid low state
557 if (a->low_count > 0 && a->state != low) {
558 a->low_count += 1;
559
560 if (a->low_count >= 20) {
561 if (a->invalid_low >= 16) {
562 a->state = low; // If 16 of last 20 samples were invalid low update state
563 a->time = millis;
564 }
565 // After 20 samples, reset the variables
566 a->invalid_low = 0;
567 a->low_count = 0;
568 }
569 }
570
571 // Handles 16/20 sample requirement for invalid high state
572 if (a->high_count > 0 && a->state != high) {
573 a->high_count += 1;
574
575 if (a->high_count >= 20) {
576 if (a->invalid_high >= 16) {
577 a->state = high; // If 16 of last 20 samples were invalid high update state
578 a->time = millis;
579 }
580 // After 20 samples, reset the variables
581 a->invalid_high = 0;
582 a->high_count = 0;
583 }
584 }
585}

601void debounce_btn(struct btn *b, uint32_t digital_condition, unsigned char btn_id) {
602 uint8_t flag = 0;
603 // Check analog switches
604 debounce_analog_contact(&(b->A1));
605 debounce_analog_contact(&(b->A2));
606
607 // Check digital switch
608 if (digital_condition == 0 && b->D1.state != press) { // If state is pressed
609 b->D1.press += 1;
610 b->D1.unpress = 0;
611 if (b->D1.press >= DL) {
612 b->D1.state = press;
613 b->D1.time = millis;
614 }
615 }
616 else if (digital_condition != 0 && b->D1.state != open) { // If state is unpressed
617 b->D1.press = 0;
618 b->D1.unpress += 1;
619 if (b->D1.unpress >= DL) {
620 b->D1.state = open;
621 b->D1.time = millis;
622 }
623 }
624
625 // Determine final debounced state based on the states of the three contacts
626 // -- Code that requires the digital contact and a single analog contact contacts in agreement
627 if (b->D1.state == press && (b->A1.state == press || b->A2.state == press) && b->state != press) {
628 b->state = press;
629 b->time = millis;
630 }
631 else if (b->D1.state == open && (b->A1.state == open || b->A2.state == open) && b->state != open) {
632 b->state = open;
633 b->time = millis;
634 }
635 else { // Used to report invalid states
636 if (b->D1.state != open && b->D1.state != press) flag++;
637 if (b->A1.state != open && b->A1.state != press) flag++;
638 if (b->A2.state != open && b->A2.state != press) flag++;
639
640 if (flag >= 2) {
641 if (b->D1.state == b->A1.state || b->D1.state == b->A2.state) {
642 b->state = b->D1.state;
643 }
644 else if (b->A1.state == b->A2.state) {
645 b->state = b->A1.state;
646 }
647 else {
648 b->state = none;
649 }
650
651 b->time = millis;
652
653 if (b->last != b->state) {
654 b->last = b->state;
655 btn_diagnostic_report(b, btn_id); // If at least 2 contacts are debounced outside of press or open states report invalid state
656 }
657 }
658 }

671void init_digital_pins(void) {
672 // Button pins
673 PCC->PCCn[PCC_PORTC_INDEX] |= 0x40000000; // Write Clock Gate Control high to allow updating registers
674 PCC->PCCn[PCC_PORTB_INDEX] |= 0x40000000;
675
676 // Set Pin Control Registers
677 PORTB->PCR[0] = 0x00000103; // Puts pin in alternate mode 1 (GPIO use), enables and sets internal pull up resistor
678 PORTC->PCR[0] = 0x00000103;
679 PORTC->PCR[14] = 0x00000103;
680 PORTC->PCR[16] = 0x00000103;
681
682 // Set Port Data Direction Register for inputs
683 PTB->PDDR &=~ (0x1);
684 PTC->PDDR &=~ ((0x1 << 16) | (0x1 << 14) | 0x1);
685
686 // 5V_GRPx_nEN pins
687 PCC->PCCn[PCC_PORTD_INDEX] |= 0x40000000;
688
689 PORTB->PCR[8] = 0x00000102; // Sets pins in GPIO mode with pulldown resistor enabled
690 PORTB->PCR[11] = 0x00000102;
691 PORTD->PCR[17] = 0x00000102;
692
693 PTB->PDDR |= (0x1 << 8) | (0x1 << 11); // Set pins as outputs
694 PTD->PDDR |= (0x1 << 17);
695
696 PTB->PSOR &=~ (0x1 << 8) | (0x1 << 11); // Set outputs low
697 PTD->PSOR &=~ (0x1 << 17);
698}
